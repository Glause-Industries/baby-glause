Backend

Setup & Instructions

Connect this editor to a Google Sheets backend so your changes publish live to the family viewer.

1
Create a new Google Sheet at sheets.google.com.
2
Open Extensions → Apps Script and paste the Code.gs content below.
// Code.gs — Baby Glause Backend // Paste this into your Apps Script editor const ADMIN_PASSWORD = 'babyglause2026'; // Change this to your desired password const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doPost(e) { try { const params = JSON.parse(e.postData.contents); const { action, password } = params; if (password !== ADMIN_PASSWORD) { return jsonResponse({ ok: false, error: 'Unauthorized' }); } if (action === 'saveConfig') { saveConfig(params.config); return jsonResponse({ ok: true }); } if (action === 'saveJournal') { saveJournal(params.entries); return jsonResponse({ ok: true }); } if (action === 'saveEvents') { saveEvents(params.raw); return jsonResponse({ ok: true }); } return jsonResponse({ ok: false, error: 'Unknown action' }); } catch (err) { return jsonResponse({ ok: false, error: err.toString() }); } } function doGet(e) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const configSheet = ss.getSheetByName('Config'); const journalSheet = ss.getSheetByName('Journal'); const eventsSheet = ss.getSheetByName('Events'); const config = configSheet ? Object.fromEntries( configSheet.getDataRange().getValues().map(r => [r[0], r[1]]) ) : {}; const journalRows = journalSheet ? journalSheet.getDataRange().getValues().slice(1) : []; const entries = journalRows.map(r => ({ date: r[0], title: r[1], body: r[2] })); const eventsRaw = eventsSheet ? (eventsSheet.getRange(1,1).getValue() || '') : ''; return jsonResponse({ ok: true, config, entries, eventsRaw }); } function saveConfig(config) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName('Config') || ss.insertSheet('Config'); sheet.clearContents(); sheet.appendRow(['welcomeText', config.welcomeText || '']); sheet.appendRow(['weeklyNote', config.weeklyNote || '']); } function saveJournal(entries) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName('Journal') || ss.insertSheet('Journal'); sheet.clearContents(); sheet.appendRow(['Date', 'Title', 'Body']); entries.forEach(e => sheet.appendRow([e.date, e.title, e.body])); } function saveEvents(raw) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName('Events') || ss.insertSheet('Events'); sheet.clearContents(); sheet.getRange(1, 1).setValue(raw); } function setupSheets() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); ['Config', 'Journal', 'Events'].forEach(name => { if (!ss.getSheetByName(name)) ss.insertSheet(name); }); Logger.log('Sheets created.'); } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
3
Run the setupSheets() function once to create the required sheets.
4
Click Deploy → New deployment. Set type to Web App. Execute as Me, access to Anyone. Click Deploy.
5
Copy the Web App URL from the deployment dialog.
6
Paste it into the API URL field below and click Test Connection.
7
Also update index.html — replace the SHEETS_API_URL placeholder with the same URL so the family viewer can fetch live data.
